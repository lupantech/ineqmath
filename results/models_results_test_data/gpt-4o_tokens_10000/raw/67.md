\[

\[
(a+b)^4 + (b+c)^4 + (c+a)^4 = 2(a^4 + b^4 + c^4) + 4(a^3b + b^3c + c^3a + ab^3 + bc^3 + ca^3) + 6(a^2b^2 + b^2c^2 + c^2a^2)
